# I implemented rule-based intent detection using keyword matching.
# The function classifies user input into greeting, pricing, high-intent,or other categories

def detect_intent(user_input):
    user_input = user_input.lower()

    if any(x in user_input for x in ["hi", "hello", "hey"]):
        return "greeting"

    elif any(x in user_input for x in ["price", "plan", "cost"]):
        return "pricing"

    elif any(x in user_input for x in ["buy", "subscribe", "try", "interested"]):
        return "high_intent"

    else:
        return "other"

# I implemented RAG by storing product information in a JSON-based knowledge structure
# and retrieving it dynamically using the get_pricing_info function.

def get_pricing_info():
    response = ""
    for plan in data["plans"]:
        response += f"\n{plan['name']} Plan: {plan['price']}\n"
        response += "Features: " + ", ".join(plan["features"]) + "\n"
    return response

# I implemented a mock lead capture function that simulates a backend API call.
# It is triggered only after collecting all required user details
def mock_lead_capture(name, email, platform):
    print(f"\n✅ Lead captured successfully: {name}, {email}, {platform}")

# I used a list to store chat history so the agent can maintain conversation context
# I used a dictionary to maintain lead data state across multiple turns,
# ensuring that the agent collects user details step-by-step.”

chat_history = []

lead_data = {
    "name": None,
    "email": None,
    "platform": None
}

# I implemented a step-by-step lead collection function
# that checks missing fields and collects them sequentially.
# Once all required data is collected, it triggers the lead capture function

def collect_lead():
    if not lead_data["name"]:
        lead_data["name"] = input("Enter your name: ")
        return

    if not lead_data["email"]:
        lead_data["email"] = input("Enter your email: ")
        return

    if not lead_data["platform"]:
        lead_data["platform"] = input("Enter your platform: ")
        return

    mock_lead_capture(
        lead_data["name"],
        lead_data["email"],
        lead_data["platform"]
    )

print("AutoStream AI Agent Started (type 'exit' to stop)\n")

while True:
    user_input = input("You: ")

    if user_input.lower() == "exit":
        break

    chat_history.append(user_input)

    intent = detect_intent(user_input)

    if intent == "greeting":
        print("Agent: Hi! How can I help you?")

    elif intent == "pricing":
        print("Agent: Here are our plans:")
        print(get_pricing_info())

    elif intent == "high_intent":
        print("Agent: Great! Let’s get you started 🚀")
        collect_lead()

    else:
        if None in lead_data.values():
            collect_lead()
        else:
            print("Agent: Can you please clarify?")
